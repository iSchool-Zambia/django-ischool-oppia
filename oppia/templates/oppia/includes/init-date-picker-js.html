function setUpDatePicker(){ startDate = $( "#id_start_date" ); if (startDate.length) //Check if not empty jQuery object startDate.datepicker({dateFormat: "yy-mm-dd", defaultDate: "-1m"}) .datepicker('setDate', startDate.val()); endDate = $( "#id_end_date" ); if (endDate.length) //Check if not empty jQuery object endDate.datepicker({dateFormat: "yy-mm-dd", defaultDate: "-1d"}) .datepicker('setDate', endDate.val()); }
